─── STATE MANAGEMENT (useSovereignStore) ────────────────────────
File: src/store/useSovereignStore.ts
Library: Zustand
Storage: localStorage (key: "smriti_workbench_draft")

Interface:
  sheets: Record<string, SheetState>     All workbench sheets
  activeSheet: string                    Active tab (default: "ITEM")
  zoomLevel: number                      Grid zoom 50-200% (default: 100)

SheetState:
  gridData: Record<string, any>          {"{row}-{col}": value} — O(1) access
  modifiedRows: Set<number>              Rows with unsaved changes
  deletedRows: Set<number>               Rows marked for deletion
  rowsCount: number                      Row count (default: 50)
  schema: string                         Schema key → SCHEMAS map

Actions:
  setActiveSheet(name)
  setZoomLevel(level)
  updateSheet(name, updates)             Auto-saves to localStorage
  addSheet(name, initialState?)
  deleteSheet(name)
  resetStore()                           Clears localStorage + state
